public class For{
  public static void main(String args[]){
    
    System.out.println("");

    for(int i = 1; i <= 5; i++){
      System.out.print(i + ", ");
    }

    System.out.println("");
    System.out.println("");

    for(int i = 2; i <= 10; i+=2){
      System.out.print(i + ", ");
    }
    System.out.println("");
 }
}
